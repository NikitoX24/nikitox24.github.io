//Instanciar y Asignar
class Person{
  public $name;
  public function speak( ) {
    echo "Hola me llamo ". $this->name;
  }
}

$p1 = new Person( ); // Instanciar
$p1->name = "David"; // Asignar
echo $p1->name; // David
$p1->speak( ); // Hola me llamo David
// Constructor - Desconstructor
class Objeto{
  public function __construct( ) {
    echo "Object created!";
  }
  public function __destruct( ) {
    echo "Object destroyed!";
  }
}

$obj1 = new Objeto( ); // Object created!
unset($obj1); // Object destroyed!

Abstractas (se hereda pero no instancia).

}

Modo Serugo + Try-Cath
include("Conexion.php");

try {
  // Preparar la consulta
  $stmt = $conexion->prepare("SELECT * FROM tabla WHERE name = ?;");

  // Vincular y Ejecutar (Modo seguro)
  $stmt->bind_param("s", "nico"); // string, int, bool,
  $stmt->execute();
  $resultado = $stmt->get_result();
  $stmt->close();
}
catch ( Exception $e) {
  http_response_code(400);
  echo json_encode(array("error" => $e)); // (no enviar al usuario!)
}
